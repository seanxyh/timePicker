	使用方式：
	第一步：导入文件
		< !-- 控制时间面板大小 -->
		< link rel="stylesheet" media="all" type="text/css" href="./src/css/tpstyle.css" />
		< !-- 控制时间面板大小 -->

		< link rel="stylesheet" media="all" type="text/css" href="./src/css/jquery-ui.css" />
		< link rel="stylesheet" media="all" type="text/css" href="./src/css/jquery-ui-timepicker-addon.css" />
		
		< script type="text/javascript" src="./src/js/jquery-1.9.1.min.js">
		< script type="text/javascript" src="./src/js/jquery-ui.min.js">
		< script type="text/javascript" src="./src/js/jquery-ui-timepicker-addon.js">
		< script type="text/javascript" src="./src/js/jquery-ui-sliderAccess.js">
		
	第二步：面板时间调整
		< script type="text/javascript">
		
			$(function(){
				$('#basic_example_1').datetimepicker({
					//时间设置
					dateFormat: "yy-dd-mm",
					separator: ' ',
					timeFormat: 'hh:mm:ss',
					
					//显示设置	
					numberOfMonths: 2,
					minDate: 0,
					maxDate: 30,
				});
			});
			
		< /script>
	
	第三步：元素放置(设置ID)
		< input type="text" name="basic_example_1" id="basic_example_1" value="" />
